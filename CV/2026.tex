\section{Selected Publications}
\runsubsection{CAGS: Color-Adaptive Volumetric Video Streaming with Dynamic 3D Gaussian Splatting}
\descript{SIGGRAPH, 2026}
\begin{tightemize}
  \item \textbf{Daheng Yin}, Yili Jin, Jianxin Shi, Isaac Ding, Miao Zhang, Fangxin Wang, Zhaowu Huang, Cong Zhang, Jiangchuan Liu, Fang Dong
  \item We propose color-adaptive 3DGS streaming for remote physical environments: compresses dynamic 3DGS volumetric video into vector-quantized LoD representations and restores color fidelity on the client, preserving photorealistic access under fluctuating bandwidth.
\end{tightemize}

\runsubsection{TrackerSplat: Exploiting Point Tracking for Fast and Robust Dynamic 3D Gaussians Reconstruction}
\descript{SIGGRAPH Asia, 2025}
\begin{tightemize}
  \item \textbf{Daheng Yin}, Isaac Ding, Yili Jin, Jianxin Shi, Jiangchuan Liu
  \item High-performance dynamic 3D scene reconstruction for fast-moving scenes with point tracking: lifts cross-frame point trajectories onto 3D Gaussians and guides their translation, rotation, and scale compensation, reducing drift, artifacts, and temporal flickering.
\end{tightemize}

\runsubsection{FSVFG: Towards Immersive Full-Scene Volumetric Video Streaming with Adaptive Feature Grid}
\descript{ACM Multimedia, 2024}
\begin{tightemize}
  \item \textbf{Daheng Yin}, Jianxin Shi, Miao Zhang, Zhaowu Huang, Jiangchuan Liu, Fang Dong
  \item Adaptive feature-grid streaming for large-scale full-scene volumetric video: represents wide-area dynamic scenes with compact feature grids and selectively transmits features and residuals, balancing scene coverage, visual quality, and transmission cost.
\end{tightemize}
